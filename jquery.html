jquery元素选择器

$("p")选取p元素

$("p.demo")选择所有class="demo"的p元素

$("p#demo")选择所有id="demo"的p元素

jquery属性选择器

$("[href]")选取所有带有href属性的元素

$("[href='#']")选取所有带有href值等于"#"的元素

$("[href!='#']")选取所有带有href值不等于"#"的元素

$("[href$='.jpg']")选取所有带有href值等于".jpg"结尾的元素
